PRACTICA 2: Prestamos de Materiales de Biblioteca

Definición de Problema:

Una biblioteca solicita un programa en el cual pueda controlar el inventario de libros, los usuarios y los prestamos para estos mismos, además sus empleados podrán acceder a los prestamos realizados. El programa debe de tener registro de sus usuarios, de sus empleados y de cada libro existente.

C O N S U L T A S


CREATE database bibliotec;


CREATE TABLE bibliotec.clientes (ID INT(5) NOT NULL , Nombre VARCHAR(30) NOT NULL , Apellidop VARCHAR(30) NOT NULL , ApellidoM VARCHAR(30) NOT NULL , Telefono INT(10) NOT NULL , Email VARCHAR(30) NOT NULL , Calle VARCHAR(30) NOT NULL , NumCasa INT(5) NOT NULL , Colonia VARCHAR(30) NOT NULL , Ciudad VARCHAR(30) NOT NULL , Estado VARCHAR(30) NOT NULL , FechaNac INT(12) NOT NULL , RFC INT(15) NOT NULL , PRIMARY KEY (ID)) ENGINE = InnoDB;

CREATE TABLE bibliotec.empleados (Id INT(5) NOT NULL , Nombre VARCHAR(20) NOT NULL , ApellidoP VARCHAR(20) NOT NULL , ApellidoM VARCHAR(20) NOT NULL , Puesto VARCHAR(20) NOT NULL , Turno VARCHAR(10) NOT NULL , Salario INT(6) NOT NULL , NumNomina INT(10) NOT NULL , Email VARCHAR(30) NOT NULL , Comision VARCHAR(5) NOT NULL , Telefono INT(10) NOT NULL , PRIMARY KEY (Id)) ENGINE = InnoDB;
CREATE TABLE bibliotec.ventas (Folio INT(10) NOT NULL , Fecha DATE NOT NULL , Hora DATE NOT NULL , MontoTotal INT(10) NOT NULL , Descuento INT(10) NOT NULL , TipoPago VARCHAR(30) NOT NULL , IdEmpleado INT(10) NOT NULL , IdCliente INT(10) NOT NULL , PRIMARY KEY (Folio)) ENGINE = InnoDB;
CREATE TABLE bibliotec.autos (NumSerie INT(10) NOT NULL , Model VARCHAR(30) NOT NULL , Anio INT(5) NOT NULL , Color VARCHAR(30) NOT NULL , Gamma VARCHAR(30) NOT NULL , Transmision VARCHAR(30) NOT NULL , Precio INT(10) NOT NULL , Extra TEXT NOT NULL ) ENGINE = InnoDB;
